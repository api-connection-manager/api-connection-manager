   2: /**
   3:  * class-wp-services-api
   4:  * 
   5:  * This class uses params stored in a module file to connect to remote services.
   6:  * It manages the storing of access_tokens and refresh_tokens in the oauth2
   7:  * spec and also allows for custom service api's to be connected to.
   8:  * 
   9:  * Plugins can then easily connect to these services and make requests.
  10:  * Developers can also easily create new modules with the params for different
  11:  * services.
  12:  * 
  13:  * Currently only one service can be requested at a time. As each service more
  14:  * than likely has its own params, even if two or more services had similar
  15:  * params they would still be individual calls made by the wp_response_*
  16:  * helper functions. Also if more than one service was requested and they
  17:  * required login, it would be unfeasible to have multpile login screens.
  18:  * 
  19:  * Requests made to this api will then check that the user is logged into that
  20:  * service. If they are not and the service is active then the developer should
  21:  * be able to call a login screen for that service. Some services require that
  22:  * a new tab be opened, some require only a popup with a form.
  23:  * 
  24:  * <b>N.B.</b>
  25:  * Although wordpress mixes and jumps between slugs and index files when working
  26:  * with plugins, for this class the index file and the slug are the same and the
  27:  * one. So the slug for a module stored at:
  28:  * api-con-mng-modules/google-login/index.php
  29:  * the slug would then be:
  30:  * google-login/index.php
  31:  * The reason for this is that the index file is the only definite unique item
  32:  * between different modules. Two developers giving two modules the same slug
  33:  * by accident could cause headaches. Using the index.php file makes this
  34:  * impossible. Don't forget to urlencode() your slugs before using them in
  35:  * forms etc. Also don't forget to urldecode() your slugs when parsing form
  36:  * submits.
  37:  *
  38:  * @todo localization
  39:  * @todo refresh token
  40:  * @todo research google's AuthSub or ClientLogin api's. Might have something
  41:  * to help building for custom services.

  43:  * @todo look into a sevices datatype/object that plugins would request and work
  44:  * of. Something like WP_Service. WP_Service->get_grant_uri or
  45:  * WP_Service->get_btn
  46:  * @todo create a DTO object to return from calls to the api. The DTO would

 416:     /**
 417:      * Builds an array of all installed services and their params.
 418:      * 
 419:      * The return array is split into active and inactive services. There are a
 420:      * lot of system calls and working with files. Overuse of this method would
 421:      * not be advised.
 422:      * 
 423:      * Currently this method is used in:
 424:      *  - API_Connection_Manager::__construct()
 425:      *  - API_Connection_Manager::get_services //only if no services loaded yet
 426:      * 
 427:      * @param boolean Default false. Whether to include downloadable modules.
 428:      * @return array 
 429:      * @subpackage api-core
 430:      */
 431:     private function _get_installed_services(){
 432:         
 433:         $wp_plugins = array();
 434:         $plugin_root = $this->dir_sub;
 435:         
 436:         /**
 437:          * Get list of plugin index files 
 438:          */
 439:         // Files in autoflow modules folder
 440:         $plugins_dir = @ opendir( $plugin_root );
 441:         $plugin_files = array();
 442:         if ( $plugins_dir ) {
 443:             while ( ( $file = readdir( $plugins_dir ) ) !== false ) {
 444:                 if ( substr( $file, 0, 1 ) == '.' )
 445:                     continue;
 446:                 if ( is_dir( $plugin_root . '/' . $file ) ) {
 447:                     $plugins_subdir = @ opendir( $plugin_root . '/' . $file );
 448:                     if ( $plugins_subdir ) {
 449:                         while ( ( $subfile = readdir( $plugins_subdir ) ) !== false ) {
 450:                             if ( substr( $subfile, 0, 1 ) == '.' )
 451:                                 continue;
 452:                             if ( substr($subfile, -4) == '.php' )
 453:                                 $plugin_files[] = "$file/$subfile";
 454:                         }
 455:                         closedir($plugins_subdir);
 456:                     }
 457:                 } else {
 458:                     if ( substr($file, -4) == '.php' )
 459:                         $plugin_files[] = $file;
 460:                 }
 461:             }
 462:             closedir($plugins_dir);
 463:         } //end list of plugin files
 464:         
 465:         if ( empty($plugin_files) )
 466:             return array(
 467:                 'active' => array(),
 468:                 'inactive' => array()
 469:             );
 470:         
 471:         /**
 472:          * Build up array of plugin[data]
 473:          */
 474:         foreach ( (array) $plugin_files as $plugin_file ) {
 475:             
 476:             if ( !is_readable("$plugin_root/$plugin_file") )
 477:                 continue;
 478: 
 479:             $plugin_data = get_plugin_data("$plugin_root/$plugin_file", false, false); //Do not apply markup/translate as it'll be cached.
 480:             
 481:             //get slug
 482:             if( @empty( $plugin_data['slug'] ) )
 483:                 $plugin_data['slug'] = $plugin_file;
 484:             
 485:             if ( empty($plugin_data['Name']) )
 486:                 continue;
 487: 
 488:             $wp_plugins[plugin_basename($plugin_file)] = $plugin_data;
 489:             
 490:             //get params and options
 491:             $params = $this->_get_params( plugin_basename($plugin_file) );
 492:             $options = $this->_get_service_options( plugin_basename($plugin_file) );
 493:             $wp_plugins[plugin_basename($plugin_file)]['params'] = $params;
 494:             $wp_plugins[plugin_basename($plugin_file)]['options'] = $options;
 495:         }//end build of plugin[data]
 496:         
 497:         /**
 498:          * Sort active|inactive and return result 
 499:          */
 500:         uasort($wp_plugins, '_sort_uname_callback');
 501:         $res = array(
 502:             'active' => array(),
 503:             'inactive' => $wp_plugins
 504:         );
 505:         
 506:         //get active/inactive service slugs
 507:         $api_options = $this->_get_options();
 508:         (is_array(@$api_options['active'])) ?
 509:             $active = $api_options['active'] :
 510:             $active = array();
 511:         (is_array(@$api_options['active'])) ?
 512:             $inactive = $api_options['inactive'] :
 513:             $inactive = array();
 514:         
 515:         //remove active plugins from list
 516:         foreach($active as $slug){
 517:             if(array_key_exists($slug, $wp_plugins)){
 518:                 $res['active'][$slug] = $wp_plugins[$slug];
 519:                 unset($res['inactive'][$slug]);
 520:             }
 521:         }
 522:         
 523:         //return result
 524:         return $res;
 525:     }

 607:              */
 608:             $i = wp_nonce_tick();
 609:             $nonce = substr(wp_hash($i . $slug . $user, 'nonce'), -12, 10);
 610:             /**
 611:              * End nonce 
 612:              */
 613:             
 614:             //set the state {$nonce}-{$slug}-{$userID}
 615:             $oauth2[ 'grant-vars' ][ 'state' ] = serialize(array( 
 616:                 $nonce,
 617:                 urlencode( $slug ),
 618:                 $user
 619:             ));
 620:             
 621:             //check options for grant-var values
 622:             if($options)
 623:                 foreach($options['app-grant-vars'] as $key => $opt)
 624:                     $oauth2['grant-vars'][$key] = $opt;
 625: 
 626:             //set the new service params
 627:             $uri = url_query_append( $uri, $oauth2[ 'grant-vars' ] );
 628:             $oauth2['grant-uri'] = $uri;
 629: 
 630:             if(is_wp_error($oauth2['grant-uri']))
 631:                 return $this->_error( $oauth2['grant-uri']->get_error_message());
 632:         }
 633:             
 634:         //try returning params
 635:         if(isset($oauth2)) return $oauth2;
 636:         elseif(isset($service)) return $service;
 637:         
 638:         //default error
 639:         return $this->_error("No params set for service {$slug}");
 640:     }

1079:      * 
1080:      * $res = new stdClass();
1081:      * $res->response = array();    //the module defined dto
1082:      * $res->slug = "";
1083:      * $res->user = "";

1086:      * @return stdClass|WP_error Returns an error if no service slug found.

1088:      */

1090:         
1091:         //vars

1095:         $res->user = "";

1102:             $vars = unserialize(stripslashes($response['state']));

1104:             $res->user = @urldecode($vars[2]);

1106:         
1107:         //what ever vars are left is the services response struct
