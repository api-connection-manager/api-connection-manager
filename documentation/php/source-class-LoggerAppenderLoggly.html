106:  * Description
107:  *
108:  *   The Syslog class is a syslog device implementation in PHP
109:  *   following the RFC 3164 rules.
110:  *
111:  *   Facility values:
112:  *      0 kernel messages
113:  *      1 user-level messages
114:  *      2 mail system
115:  *      3 system daemons
116:  *      4 security/authorization messages
117:  *      5 messages generated internally by syslogd
118:  *      6 line printer subsystem
119:  *      7 network news subsystem
120:  *      8 UUCP subsystem
121:  *      9 clock daemon
122:  *     10 security/authorization messages
123:  *     11 FTP daemon
124:  *     12 NTP subsystem
125:  *     13 log audit
126:  *     14 log alert
127:  *     15 clock daemon
128:  *     16 local user 0 (local0) (default value)
129:  *     17 local user 1 (local1)
130:  *     18 local user 2 (local2)
131:  *     19 local user 3 (local3)
132:  *     20 local user 4 (local4)
133:  *     21 local user 5 (local5)
134:  *     22 local user 6 (local6)
135:  *     23 local user 7 (local7)
136:  *
137:  *   Severity values:
138:  *     0 Emergency: system is unusable
139:  *     1 Alert: action must be taken immediately
140:  *     2 Critical: critical conditions
141:  *     3 Error: error conditions
142:  *     4 Warning: warning conditions
143:  *     5 Notice: normal but significant condition (default value)
144:  *     6 Informational: informational messages
145:  *     7 Debug: debug-level messages
146:  *
147:  *
148:  * Usage
149:  *
150:  *   require_once('syslog.php');
151:  *   $syslog = new Syslog($facility = 16, $severity = 5, $hostname = "", $fqdn= "", $ip_from = "", $process="", $content = "");
152:  *   $syslog->Send($server = "", $content = "", $timeout = 0);
153:  *
154:  *
155:  * Examples
156:  *
157:  *   Example 1
158:  *     <?php
159:  *         require_once('syslog.php');

260:     var $_severity; // 0-7
261:     var $_hostname; // no embedded space, no domain name, only a-z A-Z 0-9 and other authorized characters
262:     var $_fqdn;
263:     var $_ip_from;
264:     var $_process;
265:     var $_content;
266:     var $_msg;

271:     function Syslog($facility = 16, $severity = 5, $hostname = "", $fqdn = "", $ip_from = "", $process = "", $content = "") {
272:         $this->_msg = '';
273:         $this->_server = '127.0.0.1';
274:         $this->_port = 514;
275:         $this->_timeout = 10;
276: 
277:         $this->_facility = $facility;
278: 
279:         $this->_severity = $severity;
280: 
281:         $this->_hostname = $hostname;
282:         if ($this->_hostname == "") {
283:             if (isset($_ENV["COMPUTERNAME"])) {
284:                 $this->_hostname = $_ENV["COMPUTERNAME"];
285:             } elseif (isset($_ENV["HOSTNAME"])) {
286:                 $this->_hostname = $_ENV["HOSTNAME"];
287:             } else {
288:                 $this->_hostname = "WEBSERVER";
289:             }
290:         }
291:         $this->_hostname = substr($this->_hostname, 0, strpos($this->_hostname . ".", "."));
292: 
293:         $this->_fqdn = $fqdn;
294:         if ($this->_fqdn == "") {
295:             if (isset($_SERVER["SERVER_NAME"])) {
296:                 $this->_fqdn = $_SERVER["SERVER_NAME"];
297:             }
298:         }
299: 
300:         $this->_ip_from = $ip_from;
301:         if ($this->_ip_from == "") {
302:             if (isset($_SERVER["SERVER_ADDR"])) {
303:                 $this->_ip_from = $_SERVER["SERVER_ADDR"];
304:             }
305:         }
306: 
307:         $this->_process = $process;
308:         if ($this->_process == "") {
309:             $this->_process = "PHP";
310:         }
311: 
312:         $this->_content = $content;
313:         if ($this->_content == "") {
314:             $this->_content = "PHP generated message";
315:         }
316:     }

366:     function Send($server = "", $content = "", $timeout = 0) {
367:         if ($server != "") {
368:             $this->_server = $server;
369:         }
370: 
371:         if ($content != "") {
372:             $this->_content = $content;
373:         }
374: 
375:         if (intval($timeout) > 0) {
376:             $this->_timeout = intval($timeout);
377:         }
378: 
379:         if ($this->_facility < 0) {
380:             $this->_facility = 0;
381:         }
382:         if ($this->_facility > 23) {
383:             $this->_facility = 23;
384:         }
385:         if ($this->_severity < 0) {
386:             $this->_severity = 0;
387:         }
388:         if ($this->_severity > 7) {
389:             $this->_severity = 7;
390:         }
391: 
392:         $this->_process = substr($this->_process, 0, 32);
393: 
394:         $actualtime = time();
395:         $month = date("M", $actualtime);
396:         $day = substr("  " . date("j", $actualtime), -2);
397:         $hhmmss = date("H:i:s", $actualtime);
398:         $timestamp = $month . " " . $day . " " . $hhmmss;
399: 
400:         $pri = "<" . ($this->_facility * 8 + $this->_severity) . ">";
401:         $header = $timestamp . " " . $this->_hostname;
402: 
403:         if ($this->_msg != "") {
404:             $msg = $this->_msg;
405:         } else {
406:             $msg = $this->_process . ": " . $this->_fqdn . " " . $this->_ip_from . " " . $this->_content;
407:         }
408: 
409:         $message = substr($pri . $header . " " . $msg, 0, 1024);
410: 
411:         $fp = fsockopen("udp://" . $this->_server, $this->_port, $errno, $errstr);
412:         if ($fp) {
413:             fwrite($fp, $message);
414:             fclose($fp);
415:             $result = $message;
416:         } else {
417:             $result = "ERROR: $errno - $errstr";
418:         }
419:         return $result;
420:     }
